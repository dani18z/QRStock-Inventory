    description: Generar QR, consultar stock, y sumar y restar stock del QR.
paths:
  /generator_qr:
    get:
      tags:
        - API QR
      description: Generar código QR para un paquete
      parameters:
        - name: name
          in: query
          description: 'Nombre del producto'
          required: true
          schema:
            type: string
        - name: size
          in: query
          description: 'Talla de la variante'
          required: true
          schema:
            type: string 
        - name: color
          in: query
          description: 'Color de la variante'
          required: true
          schema:
            type: string
        - name: stock
          in: query
          description: 'Stock a introducir'
          required: true
          schema:
            type: integer
      responses:
        '200':
          description: Successful operation
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/qr'
        '401': 
          description: No credentials
        '400':
          description: Invalid input
        '404':
          description: Undocumented
        '422':
          description: Validation exception
        '500':
          description: Internal error service
